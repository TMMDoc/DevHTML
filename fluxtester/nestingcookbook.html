// Internal implementation
internal static void DoNewQuickNest (bool singlePart = false) {
    Utils.Type ('l');  // Switch to CutCAM view
    Utils.MainWin.FindFirstDescendant (e => e.ByName ("workflow")).Click ();

    var workflowElems = Utils.FindElement (() => Utils.MainWin.FindFirstChild ("WorkflowPanel"))
        .FindAllChildren ();
    var qnIcon = workflowElems.First (a => a.Name.StartsWith ("Quick-Nest"));
    var pt = qnIcon.GetClickablePoint ();

    // Delete existing layout
    if (workflowElems.FirstOrDefault (a => a.Name.StartsWith ("Layout")) != null) {
        Utils.MouseClick (pt, CONTROL);
        Utils.Type (LEFT); Utils.Type (ENTER);
    }

    // Create new quick nest
    Utils.MouseClick (pt, singlePart ? SHIFT : None);
    Thread.Sleep (500);
    Utils.Type (ESCAPE);
}

// Internal implementation
internal static void Renest (RenestOption option) {
    Utils.RunScript ("Flux.PartEditor.Current.Skin = Layout");
    Utils.FindElement (() => Utils.MainWin.FindFirstChild ("CmdBar")
        .FindFirstChild (a => a.ByName ("nesting"))).Click ();

    char key = (option) switch {
        RenestOption.RearrangeToCreateRemSheet => 'a',
        RenestOption.RearrangeToFullSheet => 'b',
        RenestOption.SpreadPartsAlongX => 'c',
        RenestOption.SpreadPartsAlongY => 'd',
        RenestOption.SpreadPartsAlongXY => 'e',
        RenestOption.Cancel or _ => 'f',
    };
    Utils.FindElement (() => Utils.MainWin.FindFirstChild ()
        .FindFirstChild (a => a.ByControlType (ControlType.MenuItem)));
    Utils.Type (key);
}

    SpreadPartsAlongX = 2,          // 'c'
    SpreadPartsAlongY = 3,          // 'd'
    SpreadPartsAlongXY = 4,         // 'e'

Recompute Method

// Default recompute (retool all elements)
Recompute ();

// Internal implementation
internal static void Recompute (RecomputeOption option = RecomputeOption.RetoolAllElements) {
    Utils.Type (ESC);

    bool skinIsLayout = Utils.RunScriptWithOutput ("var result = Flux.PartEditor.Current.Skin.ToString()")
        = "Layout";

    Utils.FindElement (() => Utils.MainWin.FindFirstChild ("CmdBar")
        .FindFirstChild (a => a.ByName ("recompute"))).Click ();

    char key = (option) switch {
        RecomputeOption.RetoolAllElements => skinIsLayout ? 'd' : 'a',
        RecomputeOption.ToolOnlyUntooledElements => skinIsLayout ? 'a' : 'b',
        RecomputeOption.Sequence => skinIsLayout ? 'b' : 'c',
        RecomputeOption.Routing => skinIsLayout ? 'c' : 'd',
        _ => skinIsLayout ? 'd' : 'e'
    };

    Utils.FindElement (() => Utils.MainWin.FindFirstChild ()
        .FindFirstChild (a => a.ByControlType (ControlType.MenuItem)));
    Utils.Type (key);
}

RecomputeOption Enum

internal enum RecomputeOption {
    RetoolAllElements = 0,              // CutCAM: 'a', Layout: 'd'
    ToolOnlyUntooledElements = 1,       // CutCAM: 'b', Layout: 'a'
    Sequence = 2,                       // CutCAM: 'c', Layout: 'b'
    Routing = 3,                        // CutCAM: 'd', Layout: 'c'
    Cancel = 4                          // CutCAM: 'e', Layout: 'd'
}

        var qnSettings = Utils.MainWin.FindFirstChild ("SheafEditor");
        var combo = qnSettings.FindFirstChild (a => a.ByControlType (ControlType.Pane))
            .FindAllChildren (a => a.ByControlType (ControlType.ComboBox))[1]

Best Practices

✅ DO: - Use DoNewQuickNest() for consistent nest creation - Always call Utils.Type (ESC) to finalize operations - Use measurement point conversion for consistent coordinates - Validate with both layout data AND visual comparison - Clean up temporary files in [FixtureCleanUp] - Use meaningful variable names for part coordinates - Add comments for non-obvious UI operations - Use helper methods for repeated patterns

❌ DON’T: - Hardcode pixel coordinates (use measurement points) - Skip Utils.MoveGlob() when dragging near dialogs - Forget Thread.Sleep() for async operations - Mix view switches without proper navigation - Create multiple quick-nests without cleanup - Skip assertions in validation - Use generic names (p, g, r, etc. without context) - Forget to close settings/dialogs with ESC

Common Pitfalls & Solutions

Glob Interference

// Problem: Glob blocks drag operation
// Solution: Move glob before dragging
Utils.MoveGlob ();
Utils.DragAndDrop (start, end);

View State Confusion

// Problem: Don't know which view is active
// Solution: Check via script
bool skinIsLayout = Utils.RunScriptWithOutput ("var result = Flux.PartEditor.Current.Skin.ToString()") = "Layout";

Timing Issues

// Problem: Operation not complete
// Solution: Add appropriate delays
Utils.MouseClick (pt);
Thread.Sleep (500);
Utils.Type (ESC);

Selection Not Persisting

// Problem: Selection lost after operation
// Solution: Re-select before validation
Utils.MouseClick (Utils.MeasurePtToPixel (new (50, 90)));
SelectPartsBy ("Similar");
CheckSelectedParts (...);
