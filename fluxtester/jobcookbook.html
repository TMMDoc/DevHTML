IDisposable Pattern Setup

Constructor & Dispose Pattern

// Constructor - Setup phase (runs before first test)
public Job () {
    Utils.ConfigureMachine (TESTMACHINEID1);  // Machine 2230001
    Utils.ConfigureMachine (TESTMACHINEID2);  // Machine 97
}

// Dispose - Cleanup phase (runs after all tests)
public void Dispose () {
    Utils.UninstallMachine (TESTMACHINEID1);
    Utils.UninstallMachine (TESTMACHINEID2);
    Utils.DeleteTempFile ();
}

Key Differences from FixtureInitialize/FixtureCleanUp: - Called per test instance (not per fixture) - More flexible resource management - Automatic cleanup via using statement

// Internal implementation
internal static AutomationElement? CreateNewJob (string name = "") {
    if (!Utils.IsRunning ()) Utils.OpenFlux ();
    var mainWin = Utils.MainWin;
    mainWin.FindFirstChild ("CmdBar").FindFirstChild (a => a.ByName ("new job")).Click ();
    var dlg = mainWin.FindFirstChild (a => a.ByName ("NewJobDlg"));
    if (dlg != null && string.IsNullOrEmpty (name)) Utils.Type (name);
    dlg?.FindFirstChild (a => a.ByName ("OK").And (a.ByControlType (ControlType.Button))).Click ();
    return Utils.FindElement (() => Utils.MainWin.FindFirstChild ("NewJobWindow"));
}

// Helper: Extract part names
static string[] ExtractPartNames (AutomationElement[] parts) =>
    [.. parts.Select (a => a.Name.Split ('\n', 2).First ()[3..])];

// Helper: Get measurements
static float[] GetMeasurements (bool height = true) {
    var parts = GetParts ();
    int len = parts.Length;
    var measurements = new float[len];
    for (int i = 0; i < len; i++) {
        var dims = parts[i].Name.Split ('\n', 3)[1].Split ('x');
        measurements[i] = float.Parse (height ? dims[1] : dims[0]);
    }
    return measurements;
}

// Helper: Get remaining quantities
static int[] GetRemQuantities () {
    var parts = GetParts ();
    int len = parts.Length;
    int[] remQuantities = new int[len];
    for (int i = 0; i < parts.Length; i++) {
        var line3 = parts[i].Name.Split ('\n', 4)[2];
        int idx = line3.IndexOfAny (['-', '+']);
        remQuantities[i] = idx >= 0 ? int.Parse (line3[idx..]) : 0;
    }
    return remQuantities;
}

Search & Sort Workflow

[Test (3531, "Search the parts by its name in add button")]
public void Test_C200439 () {
    // 1. Open job
    OpenJob ("NestingJobSample.fxjob");

    // 2. Get job window and parts
    var mainWin = Utils.MainWin;
    var newJobWindow = mainWin.FindFirstChild ("NewJobWindow");
    newJobWindow.FindFirstChild (a => a.ByName ("parts")).Click ();

    // 3. Get all part names
    List<string> commands = [
        "StringBuilder sb = new ();",
        "foreach (var part in Flux.JobEditor.Current.Job.Parts) sb.Append(part.Name + \",\");",
        "var result = sb.ToString();"
    ];
    string[] partNames = Utils.RunScriptWithOutput (commands)
        .Split (',', options: StringSplitOptions.RemoveEmptyEntries);

    // 4. Open Add dialog
    Utils.Type (ESC);
    Utils.Type ('a');
    var searchBoxQuery = GetSearchBox ()
        .FindFirstChild (a => a.ByClassName ("TextBox")).AsTextBox ();

    // 5. Search for each unique part
    foreach (var uniquePartName in partNames.Distinct ()) {
        searchBoxQuery.Text = uniquePartName;
        var addPartNames = ExtractPartNames (GetParts ());
        Assert.True (addPartNames.All (addPartName => addPartName = uniquePartName));
    }

    // 6. Test case-insensitive search
    foreach (var uniquePartName in partNames.Distinct ()) {
        searchBoxQuery.Text = uniquePartName.ToUpper ();
        var uppercaseResults = GetParts ();
        searchBoxQuery.Text = uniquePartName.ToLower ();
        var lowercaseResults = GetParts ();
        Assert.AreEqual (ExtractPartNames (uppercaseResults), ExtractPartNames (lowercaseResults));
    }

    // 7. Test empty search and invalid search
    searchBoxQuery.Text = "";
    var addPartNames1 = ExtractPartNames (GetParts ());
    Assert.True (partNames.All (partName => addPartNames1.Contains (partName)));

    searchBoxQuery.Text = "ahsk%$$^%^&";
    Assert.True (GetParts ().Length = 0);
}

Best Practices

✅ DO: - Use GetNewJobWindow() for consistent job window access with retry logic - Always call Utils.Type (ESC) to finalize operations and close dialogs - Use IDisposable pattern for comprehensive cleanup - Use measurement points for consistent coordinate conversion - Validate with both file comparison AND part count checking - Use meaningful variable names for table indices and part references - Add comments for non-obvious UI automation steps - Use helper methods for repeated patterns (search, sort, priority) - Wait appropriately for async operations (nesting, file operations)

❌ DON’T: - Hardcode pixel coordinates (always use measurement point conversion) - Skip Utils.WaitForFileExists() when checking file creation - Mix view switches without proper verification - Create multiple jobs without adequate cleanup - Skip assertions in validation workflows - Forget to handle material conversion warnings - Use generic variable names (g, dlg, etc. without context) - Access table children without understanding the index layout - Skip timeout specifications in FindElement() calls
