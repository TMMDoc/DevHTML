var descendants = element.FindAllDescendants ();

var textBoxes = element.FindAllChildren (a => a.ByClassName ("TextBox"));

Common Test Workflows

Basic Test Flow

[Test (1519, "Test Description")]
public void TestMethod () {
    // 1. Open file
    Utils.OpenFile (Utils.tData + fxFile);
    Thread.Sleep (500);

    // 2. Navigate to feature (e.g., laser tooling)
    Utils.Type ('l');

    // 3. Perform action
    LaserCutGlob (fxFile).FindFirstDescendant (a => a.ByName ("Reverse")).Click ();

    // 4. Validate result
    CheckLaserCutProfile ("C150616", "test_reversedProfile");

    // 5. Clean up (auto-handled by framework)
}

Loop-Based Test Pattern

[Test (1520, "Changing the cutting condition")]
public void C122143 () {
    Utils.OpenFile (Utils.tData + fxFile);
    foreach (var cuttingType in new[] { "Small", "Medium", "Large" }) {
        // Perform action
        ChangeCutTypeTo (cuttingType);

        // Validate each iteration
        CheckCutCondition (cuttingType);

        // Optional: Re-open for next iteration
        Utils.OpenFile (Utils.tData + fxFile);
    }
}

[Test (ID, "Settings Test")]

Multi-Step Validation Pattern

[Test (2539, "Complex Validation")]
public void Test_C145702 () {
    Utils.OpenFile (Utils.tData + "countersink.geo");

    var circlePt = Utils.MeasurePtToPixel (new (31.102, 7.319, 0));
    Utils.MouseClick (circlePt);

    // Assertion 1: Check initial state
    Assert.True (Utils.MainWin.FindFirstChild ("Glob").FindFirstChild (c => c.ByName ("Layer")) is null);

    // Perform action
    Utils.Type ('l');
    OpenLaserCAMSettings ().FindFirstChild (c => c.ByName ("Mark forming footprint")).Click ();

    // Continue workflow
    Utils.Type (ESC); Utils.Type ('r'); Utils.Type ('a');
    Utils.MouseClick (circlePt);

    // Assertion 2: Check action result
    CheckCutCondition ("Mark");

    // Assertion 3: Verify final state
    Utils.Type ('2');
    Utils.MouseClick (Utils.MeasurePtToPixel (new (31.091, 7.323, 0)));
    Assert.AreEqual (
        Utils.WaitForGlob ().FindFirstChild ("Layer").AsComboBox ().SelectedItem.Name,
        "Footprint"
    );
}

Best Practices

✅ DO: - Use descriptive test names with test case IDs - Always clean up resources in [FixtureCleanUp] - Use meaningful constants for file paths and IDs - Add comments for non-obvious UI automation steps - Use assertions with descriptive messages - Follow the test → action → validate pattern

❌ DON’T: - Use magic numbers without constants - Leave resources uncleaned - Skip assertions in loops - Use generic variable names (g, h, s, etc. without context) - Hardcode long delays without reason - Mix test logic with assertions in complex expressions

This recipe book covers all major patterns and commands used in the Fluxtester framework for creating automated UI tests for the Flux application.
